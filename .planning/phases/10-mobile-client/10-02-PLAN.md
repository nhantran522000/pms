    1. Create libs/shared-data-access/src/api/client.ts:
       ```typescript
       /**
        * Platform-aware API client
        * - Web: uses credentials: 'include' for cookie-based auth
        * - Mobile: uses Authorization header with JWT token from AsyncStorage/SecureStore
        */

       export interface ApiClientConfig {
         baseUrl: string;
         getAuthToken?: () => Promise<string | null>;
         useCookies?: boolean;
       }

       interface ApiError {
         code: string;
         message: string;
         details?: unknown;
       }

       interface ErrorResponse {
         success: false;
         error: ApiError;
       }

       async function handleResponse<T>(response: Response): Promise<T> {
         if (!response.ok) {
           let errorMessage = `HTTP ${response.status}: ${response.statusText}`;
           try {
             const errorData: ErrorResponse = await response.json();
             errorMessage = errorData.error.message || errorMessage;
           } catch {
             // If parsing fails, use default error message
           }
           throw new Error(errorMessage);
         }
         return response.json() as Promise<T>;
       }

       export async function apiRequest<T>(
         endpoint: string,
         options: RequestInit = {},
         config: ApiClientConfig
       ): Promise<T> {
         const headers: HeadersInit = {
           'Content-Type': 'application/json',
           ...options.headers,
         };

         // Mobile: Add Authorization header with JWT
         if (config.getAuthToken && !config.useCookies) {
           const token = await config.getAuthToken();
           if (token) {
             headers['Authorization'] = `Bearer ${token}`;
           }
         }

         const response = await fetch(`${config.baseUrl}${endpoint}`, {
           ...options,
           headers,
           credentials: config.useCookies ? 'include' : 'omit',
         });

         return handleResponse<T>(response);
       }

       // Factory function to create configured API methods
       export function createApiClient(config: ApiClientConfig) {
         return {
           get: <T>(url: string) =>
             apiRequest<T>(url, { method: 'GET' }, config),
           post: <T>(url: string, data: unknown) =>
             apiRequest<T>(url, { method: 'POST', body: JSON.stringify(data) }, config),
           put: <T>(url: string, data: unknown) =>
             apiRequest<T>(url, { method: 'PUT', body: JSON.stringify(data) }, config),
           del: <T>(url: string) =>
             apiRequest<T>(url, { method: 'DELETE' }, config),
         };
       }
       ```

    import { getAccounts, getAccountSummaries, getTotalBalance, getTransactions, getCategories, getCategoryTree, getBudgets } from '../api/financial';
    import { queryKeys } from '../query-keys';
    import type { ApiClientConfig } from '../api/client';
    import type { Account, Transaction, Category, Budget, TransactionFilters } from '../types/financial';

    interface UseFinancialOptions {
      config: ApiClientConfig;
      queryOptions?: Omit<UseQueryOptions, 'queryKey' | 'queryFn'>;

      return useQuery({

        ...options.queryOptions,
      });

      });

      return useQuery({

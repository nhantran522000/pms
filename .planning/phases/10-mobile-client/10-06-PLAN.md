       ```typescript
       import * as SQLite from 'expo-sqlite';
       import NetInfo, { type NetInfoState } from '@react-native-community/netinfo';
       import { api } from '../api';
       import type { QueuedMutation, MutationType, MutationHandlers } from './types';

       const SYNC_INTERVAL_MS = 5 * 60 * 1000; // 5 minutes per CONTEXT.md
       const MAX_RETRIES = 5;

       export class OfflineQueue {
         private db: SQLite.SQLiteDatabase | null = null;
         private syncInterval: ReturnType<typeof setInterval> | null = null;
         private mutationHandlers: MutationHandlers = {};
         private isProcessing = false;
         private netInfoUnsubscribe: (() => void) | null = null;

         async init(): Promise<void> {
           // Open database
           this.db = await SQLite.openDatabaseAsync('offline_queue.db');

           // Create table
           await this.db.execAsync(`
             PRAGMA journal_mode = WAL;

             CREATE TABLE IF NOT EXISTS queue (
               id TEXT PRIMARY KEY,
               mutation TEXT NOT NULL,
               payload TEXT NOT NULL,
               clientTimestamp INTEGER NOT NULL,
               serverTimestamp INTEGER,
               retryCount INTEGER DEFAULT 0,
               status TEXT DEFAULT 'pending'
             );

             CREATE INDEX IF NOT EXISTS idx_status ON queue(status);
             CREATE INDEX IF NOT EXISTS idx_timestamp ON queue(clientTimestamp);
           `);

           // Start network listener
           this.startNetworkListener();

           // Start periodic sync
           this.startPeriodicSync();

           console.log('OfflineQueue initialized');
         }

         registerMutationHandlers(handlers: MutationHandlers): void {
           this.mutationHandlers = { ...this.mutationHandlers, ...handlers };
         }

         async enqueue(
           mutation: MutationType,
           payload: unknown,
           serverTimestamp?: number
         ): Promise<string> {
           if (!this.db) {
             throw new Error('OfflineQueue not initialized');
           }

           const id = crypto.randomUUID();
           const item: QueuedMutation = {
             id,
             mutation,
             payload: JSON.stringify(payload),
             clientTimestamp: Date.now(),
             serverTimestamp,
             retryCount: 0,
             status: 'pending',
           };

           await this.db.runAsync(
             `INSERT INTO queue (id, mutation, payload, clientTimestamp, serverTimestamp, retryCount, status)
              VALUES (?, ?, ?, ?, ?, ?, ?)`,
             [
               item.id,
               item.mutation,
               item.payload,
               item.clientTimestamp,
               item.serverTimestamp ?? null,
               item.retryCount,
               item.status,
             ]
           );

           console.log(`Enqueued mutation: ${mutation} (${id})`);

           // Try immediate sync if online
           const netInfo = await NetInfo.fetch();
           if (netInfo.isConnected) {
             this.processQueue();
           }

           return id;
         }

         private startNetworkListener(): void {
           this.netInfoUnsubscribe = NetInfo.addEventListener((state: NetInfoState) => {
             if (state.isConnected) {
               console.log('Network connected, processing queue');
               this.processQueue();
             }
           });
         }

         private startPeriodicSync(): void {
           this.syncInterval = setInterval(() => {
             this.processQueue();
           }, SYNC_INTERVAL_MS);
         }

         async processQueue(): Promise<void> {
           if (!this.db || this.isProcessing) return;

           const netInfo = await NetInfo.fetch();
           if (!netInfo.isConnected) {
             console.log('Offline, skipping queue processing');
             return;
           }

           this.isProcessing = true;

           try {
             const items = await this.db.getAllAsync<QueuedMutation>(
               `SELECT * FROM queue WHERE status = 'pending' AND retryCount < ?
                ORDER BY clientTimestamp ASC LIMIT 50`,
               [MAX_RETRIES]
             );

             for (const item of items) {
               try {
                 await this.executeMutation(item);

                 // Mark as completed
                 await this.db!.runAsync(
                   `UPDATE queue SET status = 'completed' WHERE id = ?`,
                   [item.id]
                 );

                 console.log(`Completed mutation: ${item.mutation} (${item.id})`);
               } catch (error) {
                 console.error(`Failed mutation: ${item.mutation} (${item.id})`, error);

                 // Increment retry count
                 await this.db!.runAsync(
                   `UPDATE queue SET retryCount = retryCount + 1, status = 'failed' WHERE id = ?`,
                   [item.id]
                 );
               }
             }

             // Clean up completed items older than 1 day
             await this.db.runAsync(
               `DELETE FROM queue WHERE status = 'completed' AND clientTimestamp < ?`,
               [Date.now() - 24 * 60 * 60 * 1000]
             );
           } finally {
             this.isProcessing = false;
           }
         }

         private async executeMutation(item: QueuedMutation): Promise<void> {
           const handler = this.mutationHandlers[item.mutation];
           const payload = JSON.parse(item.payload);

           if (handler) {
             await handler(payload);
           } else {
             // Default API-based execution
             await this.executeApiMutation(item.mutation, payload);
           }
         }

         private async executeApiMutation(mutation: MutationType, payload: unknown): Promise<void> {
           // Map mutations to API endpoints
           const mutationToEndpoint: Record<string, { method: string; path: string }> = {
             createTransaction: { method: 'POST', path: '/financial/transactions' },
             updateTransaction: { method: 'PUT', path: '/financial/transactions' },
             deleteTransaction: { method: 'DELETE', path: '/financial/transactions' },
             createHabit: { method: 'POST', path: '/habits' },
             updateHabit: { method: 'PUT', path: '/habits' },
             deleteHabit: { method: 'DELETE', path: '/habits' },
             createHabitLog: { method: 'POST', path: '/habits/logs' },
             createTask: { method: 'POST', path: '/tasks' },
             updateTask: { method: 'PUT', path: '/tasks' },
             deleteTask: { method: 'DELETE', path: '/tasks' },
             createHealthLog: { method: 'POST', path: '/health/logs' },
             createNote: { method: 'POST', path: '/notes' },
             updateNote: { method: 'PUT', path: '/notes' },
             deleteNote: { method: 'DELETE', path: '/notes' },
             createHobby: { method: 'POST', path: '/hobbies' },
             updateHobby: { method: 'PUT', path: '/hobbies' },
             deleteHobby: { method: 'DELETE', path: '/hobbies' },
             createHobbyLog: { method: 'POST', path: '/hobbies/logs' },
           };

           const endpoint = mutationToEndpoint[mutation];
           if (!endpoint) {
             throw new Error(`Unknown mutation type: ${mutation}`);
           }

           // Handle dynamic paths (e.g., /habits/123)
           let path = endpoint.path;
           const record = payload as Record<string, unknown>;
           if (record.id && (endpoint.method === 'PUT' || endpoint.method === 'DELETE')) {
             path = `${endpoint.path}/${record.id}`;
           }

           switch (endpoint.method) {
             case 'POST':
               await api.post(path, payload);
               break;
             case 'PUT':
               await api.put(path, payload);
               break;
             case 'DELETE':
               await api.del(path);
               break;
           }
         }

         async getPendingCount(): Promise<number> {
           if (!this.db) return 0;

           const result = await this.db.getFirstAsync<{ count: number }>(
             `SELECT COUNT(*) as count FROM queue WHERE status = 'pending'`
           );

           return result?.count ?? 0;
         }

         async clearCompleted(): Promise<void> {
           if (!this.db) return;
           await this.db.runAsync(`DELETE FROM queue WHERE status = 'completed'`);
         }

         destroy(): void {
           if (this.syncInterval) {
             clearInterval(this.syncInterval);
             this.syncInterval = null;
           }

           if (this.netInfoUnsubscribe) {
             this.netInfoUnsubscribe();
             this.netInfoUnsubscribe = null;
           }

           this.db = null;
         }
       }

       export const offlineQueue = new OfflineQueue();
       ```

    ```typescript
    import NetInfo, { type NetInfoState } from '@react-native-community/netinfo';

    interface UseNetworkStatusReturn {
      isConnected: boolean | null;
      isInternetReachable: boolean | null;
      type: string | null;
      isOffline: boolean;
    }

    export function useNetworkStatus(): UseNetworkStatusReturn {
      const [state, setState] = useState<NetInfoState>({
        type: 'unknown',
        isConnected: null,
        isInternetReachable: null,
        details: null,
        isConnectionExpensive: null,
      });

      useEffect(() => {
        // Get initial state
        NetInfo.fetch().then(setState);

        // Subscribe to changes
        const unsubscribe = NetInfo.addEventListener((newState) => {
          setState(newState);
        });

        return () => {
          unsubscribe();
        };
      }, []);

      return {
        isConnected: state.isConnected,
        isInternetReachable: state.isInternetReachable,
        type: state.type,
        isOffline: state.isConnected === false,
      };
    }
    ```

    ```typescript
    import { useMutation, useQueryClient, type UseMutationOptions } from '@tanstack/react-query';
    import { offlineQueue, type MutationType } from '../lib/queue';
    import { useNetworkStatus } from './useNetworkStatus';

    interface UseOfflineMutationOptions<TData, TError, TVariables>
      extends Omit<UseMutationOptions<TData, TError, TVariables>, 'mutationFn'> {
      mutation: MutationType;
      mutationFn: (variables: TVariables) => Promise<TData>;
      queryKey?: unknown[];
      serverTimestamp?: number;
    }

    export function useOfflineMutation<TData, TError, TVariables>(
      options: UseOfflineMutationOptions<TData, TError, TVariables>
    ) {
      const { mutation, mutationFn, queryKey, serverTimestamp, ...mutationOptions } = options;
      const queryClient = useQueryClient();
      const { isConnected } = useNetworkStatus();

      return useMutation<TData, TError, TVariables>({
        ...mutationOptions,
        mutationFn: async (variables: TVariables) => {
          // If offline, queue the mutation
          if (isConnected === false) {
            await offlineQueue.enqueue(mutation, variables, serverTimestamp);

            // Return optimistic result
            // The actual mutation will happen when back online
            return { queued: true, offline: true } as unknown as TData;
          }

          // Online: execute mutation directly
          return mutationFn(variables);
        },
        onSuccess: (data, variables, context) => {
          // Invalidate related queries
          if (queryKey) {
            queryClient.invalidateQueries({ queryKey: [queryKey] });
          }

          // Call custom onSuccess if provided
          mutationOptions.onSuccess?.(data, variables, context);
        },
        onError: async (error, variables, context) => {
          // If error is network-related and we're offline, queue it
          if (isConnected === false) {
            await offlineQueue.enqueue(mutation, variables, serverTimestamp);
          }

          // Call custom onError if provided
          mutationOptions.onError?.(error, variables, context);
        },
      });
    }

    // Convenience hooks for common mutations
    export function useCreateTransaction() {
      return useOfflineMutation({
        mutation: 'createTransaction',
        mutationFn: async (data: unknown) => {
          const response = await api.post('/financial/transactions', data);
          return response.data;
        },
        queryKey: 'financial/transactions',
      });
    }

    export function useUpdateHabit() {
      return useOfflineMutation({
        mutation: 'updateHabit',
        mutationFn: async (data: { id: string; [key: string]: unknown }) => {
          const response = await api.put(`/habits/${data.id}`, data);
          return response.data;
        },
        queryKey: 'habits/habits',
      });
    }

    export function useCreateTask() {
      return useOfflineMutation({
        mutation: 'createTask',
        mutationFn: async (data: unknown) => {
          const response = await api.post('/tasks', data);
          return response.data;
        },
        queryKey: 'tasks/tasks',
      });
    }

    // Import api from lib
    import { api } from '../lib/api';
    ```

    ```typescript
    export function OfflineBanner() {

      const translateY = useRef(new Animated.Value(-50)).current;

      useEffect(() => {
        Animated.timing(translateY, {
          toValue: isOffline ? 0 : -50,
          duration: 300,

          useNativeDriver: true,

          style={{ transform: [{ translateY }] }}
        >
          <View className="flex-row items-center justify-center">
            <Text className="text-yellow-900 font-medium text-sm">
              You are offline. Changes will sync when connected.
            </Text>
          </View>
        </Animated.View>
    ```

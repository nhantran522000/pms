  <action>
    Create Babel and Metro configuration files for Expo SDK 55.

    1. Create apps/mobile/babel.config.js:
       ```javascript
       module.exports = function (api) {
         api.cache(true);
         return {
           presets: ['babel-preset-expo'],
         };
       };
       ```

    2. Create apps/mobile/metro.config.js for monorepo support:
       ```javascript
       const { getDefaultConfig } = require('expo/metro-config');
       const path = require('path');

       const projectRoot = __dirname;
       const workspaceRoot = path.resolve(projectRoot, '../..');

       const config = getDefaultConfig(projectRoot);

       // Watch all files in the monorepo
       config.watchFolders = [workspaceRoot];

       // Resolve modules from workspace root
       config.resolver.nodeModulesPaths = [
         path.resolve(projectRoot, 'node_modules'),
         path.resolve(workspaceRoot, 'node_modules'),
       ];

       module.exports = config;
       ```

    3. Create apps/mobile/app/_layout.tsx (root layout):
       ```typescript
       import { Stack } from 'expo-router';
       import { StatusBar } from 'expo-status-bar';

           <>
             <StatusBar style="auto" />

               <Stack.Screen name="index" options={{ title: 'PMS Mobile' }} />
             </Stack>
           </>
         );
       }
       ```

  <action>Create TrialWarningJob class that handles pg-boss 'trial-warning-email' jobs:

```typescript
import { Db, Job } from 'pg-boss';

@Injectable()
export class TrialWarningJob implements OnModuleInit {
  constructor(
    private readonly pgBoss: Db,
    private readonly trialWarningService: TrialWarningService,
  ) {}

  async onModuleInit() {
    await this.pgBoss.work('trial-warning-email', this.handleJob.bind(this));
  }

  private async handleJob(job: Job<{ tenantId: string; trialEndDate: string }>) {
    const { tenantId, trialEndDate } = job.data;
    await this.trialWarningService.sendTrialWarningEmail(
      tenantId,
      new Date(trialEndDate),
    );
  }
}
```

Follow existing pg-boss job patterns from feature-financial module.

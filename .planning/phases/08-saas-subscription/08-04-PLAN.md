```typescript
import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { PLAN_KEY } from '../decorators/require-plan.decorator';
import { SubscriptionTier } from '../../domain/enums/subscription-tier.enum';

@Injectable()
export class PlanFeatureGuard implements CanActivate {
  constructor(private readonly reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredTier = this.reflector.getAllAndOverride<SubscriptionTier>(
      PLAN_KEY,
      [context.getHandler(), context.getClass()],
    );

    // No decorator = no restriction (opt-in pattern per RESEARCH.md)
    if (!requiredTier) return true;

    // FREE tier requirement = accessible to all
    if (requiredTier === SubscriptionTier.FREE) return true;

    const request = context.switchToHttp().getRequest();
    const tenant = request.tenant; // Set by TenantContextMiddleware

    if (!tenant) {
      throw new ForbiddenException({
        message: 'Tenant context not found',
        upgradeUrl: this.getUpgradeUrl(),
      });
    }

    if (this.hasAccess(tenant.subscriptionTier, requiredTier)) {
      return true;
    }

    throw new ForbiddenException({
      message: 'This feature requires a PRO subscription',
      upgradeUrl: this.getUpgradeUrl(),
      currentTier: tenant.subscriptionTier,
      requiredTier,
    });
  }

  private hasAccess(currentTier: string, requiredTier: SubscriptionTier): boolean {
    // PRO can access everything
    if (currentTier === SubscriptionTier.PRO) return true;
    // FREE can only access FREE
    return requiredTier === SubscriptionTier.FREE;
  }

  private getUpgradeUrl(): string {
    return process.env.UPGRADE_URL || 'https://your-site.com/upgrade';
  }
}
```

Key behaviors per CONTEXT.md:
- Returns true if no @RequirePlan decorator (opt-in)
- Returns true if @RequirePlan('FREE') (accessible to all)
- Returns 403 with JSON body for insufficient tier
- JSON includes upgradeUrl for frontend to display

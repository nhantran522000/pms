## Gamification Design

### XP System
| Action | XP Earned |
|--------|-----------|
| Complete habit | 10 XP |
| Maintain 7-day streak | 50 XP bonus |
| Maintain 30-day streak | 200 XP bonus |
| Maintain 100-day streak | 500 XP bonus |

### Level Thresholds
| Level | XP Required |
|-------|-------------|
| 1 | 0 |
| 2 | 100 |
| 3 | 250 |
| 4 | 500 |
| 5 | 1000 |
| ... | XP = level^2 * 25 |

Level formula: `level = floor(sqrt(totalXP / 25)) + 1`

### Achievement Milestones
| Achievement | Condition |
|-------------|-----------|
| First Step | Complete first habit |
| Week Warrior | 7-day streak |
| Monthly Master | 30-day streak |
| Century Champion | 100-day streak |
| Level 5 | Reach level 5 |
| Level 10 | Reach level 10 |

       export class XP {
         private constructor(private readonly amount: number) {
           if (amount < 0) throw new Error('XP cannot be negative');
         }

         static fromNumber(value: number): XP {
           return new XP(value);
         }

         get value(): number {
           return this.amount;
         }

         add(other: XP): XP {
           return new XP(this.amount + other.amount);
         }

         toLevel(): Level {
           return Level.fromXP(this);
         }

         XPRequiredForNextLevel(): number {
           const currentLevel = this.toLevel();
           return currentLevel.xpRequiredForNext();
         }

         progressToNextLevel(): number {
           const currentLevel = this.toLevel();
           const currentLevelXP = currentLevel.xpRequired();
           const nextLevelXP = currentLevel.xpRequiredForNext();
           const progress = this.amount - currentLevelXP;
           const needed = nextLevelXP - currentLevelXP;
           return Math.min(100, Math.max(0, (progress / needed) * 100));
         }

         toString(): string {
           return this.amount.toString();
         }
       }
       ```

       export class Level {
         private constructor(private readonly level: number) {
           if (level < 1) throw new Error('Level must be at least 1');
         }

         // Level formula: level = floor(sqrt(totalXP / 25)) + 1
         static fromXP(xp: XP): Level {
           const level = Math.floor(Math.sqrt(xp.value / 25)) + 1;
           return new Level(level);
         }

         static fromNumber(level: number): Level {
           return new Level(level);
         }

         get value(): number {
           return this.level;
         }

         // XP required to reach this level
         xpRequired(): number {
           // Inverse of level formula: XP = (level - 1)^2 * 25
           return Math.pow(this.level - 1, 2) * 25;
         }

         // XP required to reach next level
         xpRequiredForNext(): number {
           return Math.pow(this.level, 2) * 25;
         }

         xpToNextLevel(currentXP: XP): number {
           return this.xpRequiredForNext() - currentXP.value;
         }

         isLevelUp(newXP: XP): boolean {
           return Level.fromXP(newXP).value > this.level;
         }

         toString(): string {
           return this.level.toString();
         }
       }
       ```

       @Injectable()
       export class GamificationService {
         private readonly logger = new Logger(GamificationService.name);

         constructor(
           private readonly userStatsRepository: UserStatsRepository,
           private readonly achievementRepository: AchievementRepository,
         ) {}

         /**
          * Award XP for habit completion and check achievements
          */
         async awardCompletionXP(tenantId: string, habitId: string, streak: number): Promise<{
           xpAwarded: number;
           newLevel: number;
           leveledUp: boolean;
           newAchievements: AchievementEntity[];
         }> {
           const stats = await this.userStatsRepository.findOrCreate(tenantId);
           const oldLevel = stats.currentLevel.value;

           // Calculate XP: base 10 + streak bonuses
           let xpToAward = 10;
           if (streak >= 7) xpToAward += 5;  // 15 XP for 7+ day streak
           if (streak >= 30) xpToAward += 10; // 25 XP for 30+ day streak
           if (streak >= 100) xpToAward += 25; // 50 XP for 100+ day streak

           // Add XP
           const updatedStats = await this.userStatsRepository.addXP(tenantId, xpToAward);
           await this.userStatsRepository.incrementHabitsCompleted(tenantId);

           // Calculate new level
           const newXP = XP.fromNumber(updatedStats.totalXP);
           const newLevel = Level.fromXP(newXP);

           // Update level if changed
           if (newLevel.value !== oldLevel) {
             await this.userStatsRepository.updateLevel(tenantId, newLevel.value);
           }

           // Check achievements
           const newAchievements = await this.checkAndAwardAchievements(
             tenantId,
             stats.id,
             habitId,
             streak,
             newLevel.value
           );

           this.logger.log(`Awarded ${xpToAward} XP to tenant ${tenantId}. Level: ${oldLevel} -> ${newLevel.value}`);

           return {
             xpAwarded: xpToAward,
             newLevel: newLevel.value,
             leveledUp: newLevel.value > oldLevel,
             newAchievements,
           };
         }

         private async checkAndAwardAchievements(
           tenantId: string,
           userStatsId: string,
           habitId: string,
           streak: number,
           level: number
         ): Promise<AchievementEntity[]> {
           const newAchievements: AchievementEntity[] = [];

           // First completion achievement
           const stats = await this.userStatsRepository.findByTenantId(tenantId);
           if (stats && stats.habitsCompleted === 1) {
             const achievement = await this.awardIfNotExists(
               tenantId, userStatsId, 'FIRST_COMPLETION', habitId
             );
             if (achievement) newAchievements.push(achievement);
           }

           // Streak achievements
           if (streak >= 7) {
             const achievement = await this.awardIfNotExists(
               tenantId, userStatsId, 'STREAK_7', habitId
             );
